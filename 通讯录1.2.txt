void swap(void* x,void* y,int size)
{
	char* a=(char*)x;
	char* b=(char*)y;
	while(a<(char*)x+size)
	{
		*a^=*b;
		*b^=*a;
		*a^=*b;
		a++;
		b++;
	}
}
void sort(str_p* p,str_p* q)
{
	
	while(p<q)
	{
		str_p* start=p;
		int count=0;
		while(start<q)
		{
			if(strcmp(start->name,(start+1)->name)>0)
			{
				count=1;
				swap(start,start+1,sizeof(str_p));
			}
			
			start++;
		}
		if(count==0)
		{
			break;
		}
		q--;
	}
}
