//1--push|next   -1---pop|out    0---pop|next
int PriorityCmp(char str1,char str2)
{
	if(str1=='#')
		return 1;

	if(str1=='(')
	{
		if(str2==')')
			return 0;
		else
			return 1;
	}
	if(str1=='*'||str1=='/')
	{
		if(str2=='(')
			return 1;
		else
			return -1;
	}
	if(str1=='+'||str1=='-')
	{
		if(str2=='+'||str2=='-')
			return -1;
		else if(str2==')')
			return -1;
		else if(str2=='(')
			return 1;
		else
			return 1;
	}
	
}

bool IsOperator(char str)
{
	char op[]="+-*/()";
	for(int i=0;i<strlen(op);i++)
	{
		if(str==op[i])
			return true;
	}
	return false;
}

char* GetRPN(char* str)
{
	stack<char> s;
	s.push('#');
	size_t len=strlen(str);
	char* dst=new char[2*len];
	int j=0;

	for(int i=0;i<strlen(str);i++)
	{

		if(!IsOperator(str[i]))
		{
			dst[j]=str[i];
			j++;
		}
		else
		{
			
			char ch=s.top();
			if(PriorityCmp(ch,str[i])==1)
			{
				s.push(str[i]);
			}
			if(PriorityCmp(ch,str[i])==0)
			{
				s.pop();
			}
	
			if(PriorityCmp(ch,str[i])==-1)
			{
				dst[j]=ch;
				s.pop();
				i--;
				j++;
			}
		}
	}

	char c;
	while((c=s.top())!='#')
	{
		dst[j]=c;
		j++;
		s.pop();
	}
	dst[j]='\0';
	return dst;
}

int CalculatorRPN(char* str)
{
	char* dst=GetRPN(str);
	cout<<dst<<endl;
	stack<int> s;

	for(int i=0;i<strlen(dst);i++)
	{
		if(!IsOperator(dst[i]))
		{
			s.push(dst[i]-'0');
		}
		else
		{
			int ch1;
			int ch2;
			int ret;
			ch2=s.top();
			s.pop();
			ch1=s.top();
			s.pop();
			switch(dst[i])
			{
			case '+':ret=ch1+ch2;break;
			case '-':ret=ch1-ch2;break;
			case '*':ret=ch1*ch2;break;
			case '/':
				{
					if(ch2==0)
						abort();
					else
						ret=ch1/ch2;
				}break;
			default:break;
			}
			s.push(ret);
		}
	}
	
	return s.top();

}
